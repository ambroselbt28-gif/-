<script>
    let initialSettings = {};
    let currentSettings = {};

    function SafeGetValue(obj, path, defaultValue = "--") {
        const value = path.split('.').reduce((acc, part) => acc && acc[part], obj);
        if (value === undefined || value === null) return defaultValue;
        return Array.isArray(value) ? value[0] : value;
    }

    function switchTab(event, tabName) {
        document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
        document.querySelectorAll('.tab-button').forEach(btn => btn.classList.remove('active'));
        document.getElementById(tabName).classList.add('active');
        event.currentTarget.classList.add('active');
    }

    function updateSettingSelection(group, value) {
        const container = document.querySelector(`.setting-item[data-group="${group}"]`);
        if (!container) return;
        container.querySelectorAll('.option-btn').forEach(btn => {
            btn.classList.toggle('selected', btn.dataset.value === value);
        });
        currentSettings[group] = value;
    }

    function applySettings() {
        let changes = [];
        const mapping = {
            cup: '罩杯', bodyType: '身材', mouthMode: '口腔模式', pussyMode: '小穴模式',
            anusMode: '菊穴模式', virginity: '处女膜', personality: '性格'
        };

        for (const key in currentSettings) {
            if (currentSettings.hasOwnProperty(key) && initialSettings[key] !== currentSettings[key]) {
                changes.push(`${mapping[key]}:${currentSettings[key]}`);
            }
        }

        if (changes.length > 0) {
            const inputPrompt = `<Request: 取出控制终端，更改诺娃的设定为 ${changes.join(' ')}>`;
            const inputElement = parent.document.getElementById('send_textarea');
            if (inputElement) {
                inputElement.value = inputPrompt;
                inputElement.focus();
                inputElement.dispatchEvent(new Event('input', { bubbles: true }));
            }
        }
    }

    async function initDisplay() {
        try {
            const messages = await getChatMessages(getCurrentMessageId());
            const gameData = messages?.[0]?.data?.stat_data;

            if (!gameData) {
                console.error("无法加载状态数据");
                return;
            }

            document.getElementById('avatar').src = 'https://files.catbox.moe/zhq382.png';

            document.getElementById('location').innerText = `🌍 ${SafeGetValue(gameData, '世界.地点')}`;
            document.getElementById('time').innerText = `⏱️ ${SafeGetValue(gameData, '世界.时间')}`;
            const progress = parseInt(SafeGetValue(gameData, '诺娃.状态.人格模块修复进度', '0'));
            document.getElementById('personality-progress-value').innerText = progress;
            document.getElementById('personality-progress-bar').style.width = `${(progress / 200) * 100}%`;

            document.getElementById('status-action').innerText = `行动：${SafeGetValue(gameData, '诺娃.状态.行为')}`;

            document.getElementById('status-attire').innerText = SafeGetValue(gameData, '诺娃.状态.装扮');
            document.getElementById('status-body').innerText = SafeGetValue(gameData, '诺娃.状态.身体状态');
            document.getElementById('status-mouth').innerText = SafeGetValue(gameData, '诺娃.状态.口腔状态');
            document.getElementById('status-chest').innerText = SafeGetValue(gameData, '诺娃.状态.胸部状态');
            document.getElementById('status-pussy').innerText = SafeGetValue(gameData, '诺娃.状态.小穴状态');
            document.getElementById('status-anus').innerText = SafeGetValue(gameData, '诺娃.状态.菊穴状态');

            initialSettings = {
                cup: SafeGetValue(gameData, '诺娃.配置.罩杯'),
                bodyType: SafeGetValue(gameData, '诺娃.配置.身材'),
                mouthMode: SafeGetValue(gameData, '诺娃.配置.口腔模式'),
                pussyMode: SafeGetValue(gameData, '诺娃.配置.小穴模式'),
                anusMode: SafeGetValue(gameData, '诺娃.配置.菊穴模式'),
                virginity: SafeGetValue(gameData, '诺娃.配置.处女膜'),
                personality: SafeGetValue(gameData, '诺娃.配置.性格')
            };

            for(const key in initialSettings) {
                updateSettingSelection(key, initialSettings[key]);
            }

            const personalitySetting = document.getElementById('personality-setting');
            if (progress < 200) {
                personalitySetting.classList.add('disabled');
            } else {
                personalitySetting.classList.remove('disabled');
            }

        } catch (error) {
            console.error("状态栏加载出错:", error);
            document.querySelector('.nova-panel').innerHTML = `<p style='color:red;padding:15px;'>状态栏加载出错: ${error.message}<br>请发送一条消息后再试。</p>`;
        }
    }

    document.querySelectorAll('.setting-options').forEach(group => {
        group.addEventListener('click', (event) => {
            const button = event.target.closest('.option-btn');
            const parentItem = event.target.closest('.setting-item');
            if (button && !parentItem.classList.contains('disabled')) {
                const groupName = parentItem.dataset.group;
                updateSettingSelection(groupName, button.dataset.value);
            }
        });
